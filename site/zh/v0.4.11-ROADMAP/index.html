    v0.4.11 Roadmap — Release Engineering Fix (Thorough)¶

    Status: Ready for execution. Created 2026-06-21 as preparation for next session. Theme: Thoroughly fix the 3 P0 release bugs from v0.4.8/9/10, plus add end-to-end acceptance test to prevent recurrence. NOT a band-aid — a structural fix within v0.4.x scope. Type: PATCH (release engineering fix, no API change). Predecessor: v0.4.10 (binary install works on Debian 12+, wheel still broken via UNKNOWN-0.0.0). User decision: Solve in v0.4.11, not v0.5.. Defer manylinux/ARM64/macOS wheel to v0.5+. Source issues*: PAR-0qf (P0), PAR-8cs (P0), PAR-cog (P0), PAR-j8i (P0 epic).

    Rationale¶

    Three consecutive P0 release bugs in 47 hours (PAR-0qf, PAR-8cs, PAR-cog) exposed a systemic gap: no end-to-end acceptance test gates the PyPI upload. Each individual fix was small, but each fix uncovered the next bug because the pipeline had no "prove it works before shipping" step.

    v0.4.11 closes this gap within the v0.4.x series: - All 3 known P0 root causes fixed structurally (not just locally) - New acceptance workflow runs install + import + Runtime init on 3 target platforms BEFORE any twine upload - Local dry-run script catches future regressions pre-push - v0.5+ stretch goals (manylinux glibc 2.17, ARM64, macOS wheel, OIDC) deferred

    scripts/release-acceptance-test.py¶

    #!/usr/bin/env python3
"""End-to-end install test for par-runtime wheel.

Usage: release-acceptance-test.py <wheel-path> <expected-version>

Exit codes:
  0: all tests passed
  1: Level 1 failed (import / version mismatch) — BLOCKING
  2: Level 2 failed (Runtime init / tool registration) — BLOCKING
  3: invalid arguments
"""
import json
import subprocess
import sys
import venv
from pathlib import Path


TEST_CONFIG = json.dumps({
    "persistence": ["Sqlite", ":memory:"],
    "event_bus": {
        "buffer_capacity": 10,
        "delivery": {
            "max_delivery_attempts": 3,
            "initial_retry_delay": 0.1,
            "retry_backoff": ["Fixed", 0.5],
            "delivery_timeout": 5.0,
        },
        "dlq_enabled": False,
        "critical_event_types": [],
    },
    "default_quota": {
        "max_concurrent_tasks": 4,
        "max_concurrent_tools_per_agent": 2,
        "max_tokens_per_turn": None,
        "max_total_tokens": None,
    },
    "shutdown": {
        "drain_timeout": 5.0,
        "cancel_grace_period": 2.0,
        "flush_batch_size": 100,
    },
    "llm_providers": [],
    "eval_limits": {"max_depth": 10, "max_node_visits": 1000},
    "parallel_tool_execution": True,
})


def run_in_venv(venv_python: str, code: str) -> tuple[int, str, str]:
    """Run Python code in venv, return (exit_code, stdout, stderr)."""
    result = subprocess.run(
        [venv_python, "-c", code],
        capture_output=True, text=True
    )
    return result.returncode, result.stdout, result.stderr


def main():
    if len(sys.argv) != 3:
        print(f"Usage: {sys.argv[0]} <wheel-path> <expected-version>", file=sys.stderr)
        return 3

    wheel_path = Path(sys.argv[1]).resolve()
    expected_version = sys.argv[2]

    if not wheel_path.exists():
        print(f"FAIL: wheel not found at {wheel_path}", file=sys.stderr)
        return 3

    print(f"Testing wheel: {wheel_path.name}")
    print(f"Expected version: {expected_version}")
    print(f"Wheel size: {wheel_path.stat().st_size:,} bytes")
    print()

    # Create fresh venv
    venv_dir = Path("/tmp/par-acceptance-venv")
    if venv_dir.exists():
        import shutil
        shutil.rmtree(venv_dir)
    print(f"Creating venv at {venv_dir}...")
    venv.create(venv_dir, with_pip=True, clear=True)
    venv_python = str(venv_dir / "bin" / "python")
    venv_pip = str(venv_dir / "bin" / "pip")

    # Install wheel
    print(f"\n=== Installing wheel ===")
    result = subprocess.run(
        [venv_pip, "install", "--no-cache-dir", str(wheel_path)],
        capture_output=True, text=True
    )
    if result.returncode != 0:
        print(f"FAIL: pip install failed", file=sys.stderr)
        print(result.stderr, file=sys.stderr)
        return 1
    print(f"PASS: pip install")

    # Level 1: import + version
    print(f"\n=== Level 1: import + version (MUST pass) ===")
    code = f"import par_runtime; assert par_runtime.__version__ == '{expected_version}', f'version mismatch: {{par_runtime.__version__}} != {expected_version}'; print('import ok, version', par_runtime.__version__)"
    rc, out, err = run_in_venv(venv_python, code)
    if rc != 0:
        print(f"FAIL: Level 1", file=sys.stderr)
        print(f"stdout: {out}", file=sys.stderr)
        print(f"stderr: {err}", file=sys.stderr)
        return 1
    print(f"PASS: {out.strip()}")

    # Level 2: Runtime init + tool registration + close
    print(f"\n=== Level 2: Runtime init + register_tool + close (SHOULD pass) ===")
    code = f"""
import json
from par_runtime import Runtime
config = {TEST_CONFIG!r}
rt = Runtime(config)
rt.register_tool('echo', 'Echo tool', '{{"type": "object"}}')
rt.close()
print('Runtime lifecycle ok')
"""
    rc, out, err = run_in_venv(venv_python, code)
    if rc != 0:
        print(f"FAIL: Level 2", file=sys.stderr)
        print(f"stdout: {out}", file=sys.stderr)
        print(f"stderr: {err}", file=sys.stderr)
        return 2
    print(f"PASS: {out.strip()}")

    # Level 3: informational
    print(f"\n=== Level 3: informational ===")
    code = "import par_runtime; print('exports:', par_runtime.__all__)"
    rc, out, _ = run_in_venv(venv_python, code)
    print(out.strip())

    code = "from par_runtime._ffi import _find_library; print('so path:', _find_library())"
    rc, out, _ = run_in_venv(venv_python, code)
    print(out.strip())

    print(f"\n=== ALL TESTS PASSED ===")
    return 0


if __name__ == "__main__":
    sys.exit(main())

    .github/workflows/release-acceptance.yml¶

    name: Release acceptance test

# Triggers AFTER pypi-publish.yml completes (wheel is on GH Release).
# If this workflow fails, the wheel MUST NOT be uploaded to PyPI.

on:
  workflow_run:
    workflows: ["PyPI publish"]
    types: [completed]

permissions:
  contents: write
  pull-requests: write

jobs:
  accept:
    name: Accept on ${{ matrix.container }}
    runs-on: ubuntu-latest
    container: ${{ matrix.container }}
    strategy:
      fail-fast: false
      matrix:
        container:
          - debian:12
          - ubuntu:22.04
          - ubuntu:24.04
    if: ${{ github.event.workflow_run.conclusion == 'success' }}
    steps:
      - uses: actions/checkout@v4

      - name: Install Python
        run: |
          apt-get update -qq
          apt-get install -y -qq python3 python3-pip python3-venv curl
          python3 --version

      - name: Download wheel from GH Release
        run: |
          TAG="${{ github.event.workflow_run.head_branch }}"
          echo "Downloading wheel from tag $TAG"
          curl -fsSL -o /tmp/wheel.whl \
            "https://github.com/${{ github.repository }}/releases/download/$TAG/par_runtime-$TAG-py3-none-any.whl" || true
          # The filename may have dashes (version normalization); fallback to globbing
          if [ ! -f /tmp/wheel.whl ]; then
            gh release download "$TAG" --repo "${{ github.repository }}" --pattern 'par_runtime-*.whl' --dir /tmp/
            mv /tmp/par_runtime-*.whl /tmp/wheel.whl
          fi
          ls -la /tmp/wheel.whl
        env:
          GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}

      - name: Run acceptance test
        run: |
          TAG="${{ github.event.workflow_run.head_branch }}"
          # Strip leading 'v' from tag, replace dashes appropriately for version
          VERSION="${TAG#v}"
          python3 scripts/release-acceptance-test.py /tmp/wheel.whl "$VERSION"

  gate:
    name: All acceptance tests passed
    needs: accept
    runs-on: ubuntu-latest
    if: always()
    steps:
      - name: Check matrix result
        run: |
          echo "Matrix conclusion: ${{ needs.accept.result }}"
          if [ "${{ needs.accept.result }}" != "success" ]; then
            echo "FAIL: release acceptance test failed on at least one platform"
            echo "DO NOT upload to PyPI until this is green"
            exit 1
          fi
          echo "PASS: all platforms accepted the wheel"
          echo "Next step: manual twine upload (see docs/rules/release.md)"

    scripts/release-local-test.sh¶

    #!/usr/bin/env bash
# Local dry-run of the release acceptance test.
# Requires Docker. Builds wheel locally, then runs acceptance test in each container.
set -euo pipefail

cd "$(dirname "$0")/.."

if ! command -v docker >/dev/null 2>&1; then
    echo "ERROR: docker not available. Install Docker or use CI workflow instead."
    exit 1
fi

# 1. Build wheel locally
echo "=== Building wheel locally ==="
make install-dev >/dev/null
mkdir -p bindings/python/par_runtime/lib
cp -f _build/default/lib/ffi/par_capi.so bindings/python/par_runtime/lib/
(cd bindings/python && rm -rf dist && pip wheel . -w dist/ --no-deps)
WHEEL=$(ls bindings/python/dist/par_runtime-*.whl | head -1)
VERSION=$(sed -n 's/^(version "\([^"]*\)").*/\1/p' dune-project)
echo "Wheel: $WHEEL"
echo "Version: $VERSION"

# 2. Run acceptance test in each container
CONTAINERS=(debian:12 ubuntu:22.04 ubuntu:24.04)
ALL_PASS=true

for c in "${CONTAINERS[@]}"; do
    echo ""
    echo "=== Testing on $c ==="
    if docker run --rm \
        -v "$PWD:/work" \
        -w /work \
        "$c" \
        bash -c "apt-get update -qq && apt-get install -y -qq python3 python3-pip python3-venv > /dev/null && python3 scripts/release-acceptance-test.py /work/$WHEEL $VERSION"; then
        echo "PASS: $c"
    else
        echo "FAIL: $c"
        ALL_PASS=false
    fi
done

echo ""
if $ALL_PASS; then
    echo "=== ALL PLATFORMS PASSED ==="
    exit 0
else
    echo "=== ONE OR MORE PLATFORMS FAILED ==="
    exit 1
fi
